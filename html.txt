    const prizes = [
      "🥳 خصم 50% على الكفرات!",
      "📱 غلاف مجاني عند الشراء!",

    let serialNumber = Math.floor(Math.random() * 1000000);

    function playGame() {
      if (playedToday) {
        document.getElementById("result").textContent = "⚠️ جربت اليوم! ارجع باجر للمحاولة من جديد.";
        return;
      }

      const prize = prizes[Math.floor(Math.random() * prizes.length)];
      document.getElementById("result").innerHTML = `🎊 ${prize}<br>🔢 رقم جائزتك: <strong>${serialNumber}</strong>`;
      if (!prize.includes("لا شيء")) {
        document.getElementById("winSound").play();
      }
      localStorage.setItem('baniphone_played', true);
      document.getElementById("shareBtn").style.display = "inline-block";
    }

    function shareOnWhatsApp() {
      const message = encodeURIComponent(`🎉 ربحت جائزة من مركز باني فون! رقم الجائزة: ${serialNumber}. جرب حظك انت هم: https://baniphone-game.com`);
